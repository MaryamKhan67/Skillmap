    <script>
    document.addEventListener('DOMContentLoaded', function() {
        // Create stars background
        function createStars() {
            const container = document.getElementById('stars-container');
            const starCount = 150;
            for (let i = 0; i < starCount; i++) {
                const star = document.createElement('div');
                star.classList.add('star');
                const size = Math.random() * 2 + 1;
                star.style.width = star.style.height = `${size}px`;
                star.style.left = `${Math.random() * 100}%`;
                star.style.top = `${Math.random() * 100}%`;
                star.style.animationDelay = `${Math.random() * 5}s`;
                container.appendChild(star);
            }
        }
        
        createStars();

        // Available skills by category
        const skillsByCategory = {
            frontend: ["HTML", "CSS", "JavaScript", "React", "Vue", "Angular", "TypeScript", "Sass", "Responsive Design"],
            backend: ["Node.js", "Express.js", "Python", "Django", "Flask", "Ruby", "Rails", "PHP", "Laravel", "Databases", "SQL", "MongoDB", "REST APIs", "GraphQL"],
            devops: ["Git", "Docker", "Kubernetes", "CI/CD", "AWS", "Azure", "Linux", "Bash", "Networking"],
            design: ["Figma", "Photoshop", "Illustrator", "UI Design", "UX Design", "Wireframing", "Prototyping"],
            data: ["Python", "R", "SQL", "Pandas", "NumPy", "Data Visualization", "Machine Learning", "TensorFlow", "PyTorch"]
        };
        
        let selectedRole = "";
        let selectedSkills = [];
        
        // Populate skills container with all available skills
        const skillsContainer = document.getElementById('skills-container');
        const allSkills = [...new Set(Object.values(skillsByCategory).flat())];
        
        allSkills.forEach(skill => {
            const skillElement = document.createElement('div');
            skillElement.className = 'skill-option bg-gray-700 hover:bg-purple-600 py-2 px-4 rounded-lg transition-colors text-center font-["Exo_2"]';
            skillElement.textContent = skill;
            skillElement.dataset.skill = skill;
            
            skillElement.addEventListener('click', function() {
                this.classList.toggle('bg-purple-600');
                this.classList.toggle('selected');
                
                if (this.classList.contains('selected')) {
                    selectedSkills.push(skill);
                } else {
                    selectedSkills = selectedSkills.filter(s => s !== skill);
                }
            });
            
            skillsContainer.appendChild(skillElement);
        });
        
        // Role selection
        document.querySelectorAll('.role-btn').forEach(btn => {
            btn.addEventListener('click', function() {
                // Remove selected class from all buttons
                document.querySelectorAll('.role-btn').forEach(b => {
                    b.classList.remove('bg-cyan-600');
                    b.classList.add('bg-gray-700');
                });
                
                // Add selected class to clicked button
                this.classList.remove('bg-gray-700');
                this.classList.add('bg-cyan-600');
                
                selectedRole = this.dataset.role;
                document.getElementById('targetRole').value = selectedRole;
            });
        });
        
        // Generate constellation
        document.getElementById('generate-constellation').addEventListener('click', function() {
            if (!selectedRole) {
                alert('Please select a target role.');
                return;
            }
            
            if (selectedSkills.length === 0) {
                if (!confirm('You haven\'t selected any skills. Continue anyway?')) {
                    return;
                }
            }
            
            // Store user profile in localStorage
            const userProfile = {
                targetRole: selectedRole,
                knownSkills: selectedSkills
            };
            
            localStorage.setItem('userProfile', JSON.stringify(userProfile));
            
            // Redirect to constellation page
            window.location.href = 'constellation.html';
        });
    });
    </script>
